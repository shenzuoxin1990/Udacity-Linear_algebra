def matxMultiply(A, B):
    rA,cA = shape(A)
    rB,cB = shape(B)
    if cA != rB:
        raise ValueError

    transB = transpose(B)
    result = []
    for i in range(rA):
        row = []
        for j in range(cB):
            row.append(sum([a*b for a,b in zip(A[i],transB[j])]))
        result.append(row)
    return result

""" Gaussian Jordan  Ax = b.
    
        A:  
        b: 
        decPts: 4
        epsilon: 0 1.0e-16
        
     x  Ax = b 
    None Ab 
    None A 
"""
def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    rA,cA = shape(A)
    rb,cb = shape(b)
    # None Ab 
    if rA != rb:
        return None
    Ab = augmentMatrix(A,b)
    row,col = shape(Ab)
    for c in range(col-1):
        max_array = []
        # c  c~N
        for n in range(c,row):    
            max_array.extend([abs(Ab[n][c])])
        if len(max_array) == 0:
            break
            
        # None A 
        max_c = max(max_array)
        if abs(max_c) < epsilon:
            return None
        
        # c 
        for n in range(c,row):
            if max_c == abs(Ab[n][c]):
                swapRows(Ab, c, n)
                break
                
        # cc,c1
        if abs(Ab[c][c]) < epsilon:
            return None
        devide_parameter = 1./(Ab[c][c])
        scaleRow(Ab, c, devide_parameter)

        # c0
        for i in range(row):
            if i == c:
                continue
            n = Ab[i][c]
            if abs(n) > epsilon:
                addScaledRow(Ab, i, c, (-n))
        
    # Ab
    matxRound(Ab, decPts)
    # 
    result = [[n[col-1]] for n in Ab]
    return result

# TODO MSE
def calculateMSE2D(X,Y,m,b):
    mse = 0
    n = len(X)
    for i in range(n):
        x = X[i]
        y = Y[i]
        mse += (y-(m*x)-b)**2
    mse = 1.0/n * mse
    return mse

def linearRegression2D(X,Y):
    X = [[x,1] for x in X]
    Y = [[y] for y in Y]
    x_trans = transpose(X)
